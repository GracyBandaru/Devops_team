pipeline {
    agent any
    stages {
        stage('git repo & clean') {
            steps {
               bat "rmdir  /s /q my-app"
                bat "git clone https://github.com/GracyBandaru/Devops_team.git"
                bat "mvn clean -f my-app"
            }
        }
        stage('install') {
            steps {
                bat "mvn install -f my-app"
            }
        }
        stage('test') {
            steps {
                bat "mvn test -f my-app"
            }
        }
        stage('package') {
            steps {
                bat "mvn package -f my-app"
            }
        }
    }
}
